$$\begin{aligned}5^1+5&<5^{1+1}\\
	10&<5^2\\

$$\begin{aligned}
	5^{k+1}+5&<5^{k+1+1}\\
	5^k+5&<5^{k+1}\cdot5\\
	5(5^k+1)&<5(5^{k+1})\\
	5^k+1&<5^{k+1}\\
	\text{IS:}\quad5^k+5&<5^{k+1}\\
	&\qquad\qquad\square
	\end{aligned}$$
